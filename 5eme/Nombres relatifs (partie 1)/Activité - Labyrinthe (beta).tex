\begin{luacode}
	function print_items(points)
		for _, p in ipairs(points) do
			tex.sprint("\\item (", p.x, "; ", p.y, ")")
		end
	end

	function print_path(points)
		tex.sprint("\\draw[red,thick] ")
		for i, p in ipairs(points) do
		 	if i ~= 1 then
		 		tex.sprint(" -- ")
		 	end
		 	tex.sprint("(", p.x * 2 + 3.5, ",", p.y * 2 + 2.5, ")")
		end
		tex.sprint(";")
	end
\end{luacode}
